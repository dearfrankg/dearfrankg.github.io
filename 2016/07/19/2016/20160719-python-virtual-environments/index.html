Install pyenv

brew update
brew install pyenv

// add to .bash_profile
# enable python pyenv
if which pyenv > /dev/null; then eval "$(pyenv init -)"; fi

Install python version

// see versions
$ pyenv versions
* system (set by /Users/frankg/.pyenv/version)

// install version 3.5.2
pyenv install 3.5.2

// see versions
$ pyenv versions
* system (set by /Users/frankg/.pyenv/version)
  3.5.2

globally configure a python version

// see global config
$ pyenv global
system

// set global config
$ pyenv global 3.52
$ pyenv global
3.5.2

// set global config
$ pyenv global system
$ pyenv global
system

// there is no unset

// see python path
pyenv which python
/Users/frankg/.pyenv/versions/3.5.2/bin/python

configure directory to use version

// see directory config
$ pyenv local
pyenv: no local version configured for this directory

// set directory config
$ pyenv local 3.5.2
$ pyenv local
3.5.2

// unset directory config
$ pyenv local --unset
$ pyenv local
pyenv: no local version configured for this directory

configure shell to use version

// see shell config
$ pyenv shell
pyenv: no shell-specific version configured

// set shell config
$ pyenv shell 3.5.2
$ pyenv shell
3.5.2

// unset shell config
$ pyenv shell --unset
$ pyenv shell
pyenv: no shell-specific version configured
